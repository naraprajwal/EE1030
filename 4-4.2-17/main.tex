\textbf{Question}:\\
Find the direction and the normal vector if $y=x+2$
\\ \textbf{Solution: }\\
for $y=mx+c$
    \begin{table}[h!]

Camparing the equation,
\begin{align}
    y=mx+c \ \text{and} \ y=x+2 \\
    m=1 \text \ {and} \ c=2
\end{align}
Hence direction vector is,
\begin{align}
    d = \myvec{1 \\ m}=\myvec{1 \\ 1} 
\end{align}
and normal vector is,
\begin{align}
    n = \myvec{-m \\ 1}=\myvec{-1 \\ 1}
\end{align}
